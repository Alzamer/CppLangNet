```cpp
#include <iostream>
#include <vector>
#include <algorithm>

int main() {
	std::vector<int> numbers{3, 12, 7, 18, 12};
	// Print
	for (int num : numbers) std::cout << num << ", ";

	numbers.push_back(13); // add number 13 to the end
	// Erase values 12
	std::erase(numbers.begin(), numbers.end(), 12); // C++20
	// Print
	for (int num : numbers) std::cout << num << ", ";
}
```

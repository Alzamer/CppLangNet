```cpp
#include <iostream>
#include <vector>
#include <ranges>
#include <algorithm>
namespace rg = std::ranges;

int main() {
	std::vector numbers{3, 12, 7, 18, 12};  // C++17 - type int deduced
	// Print
	for (int num : numbers) std::cout << num << ", ";

	numbers.push_back(13); // add number 13 to the end
	// Erase values 12
	rg::erase(numbers, 12); // C++20
	// Print
	for (int num : numbers) std::cout << num << ", ";
}
```

Note: `ranges` are available since **C++20**.
